# PersistentVolume
 
apiVersion: v1
kind: PersistentVolume
metadata:
  name: my-vol
spec:
  capacity:
    storage: 10Gi
  volumeMode: Filesystem
  accessModes:
      - ReadWriteOnce
  storageClassName: manual
  local:
    path: /mnt/silo
  nodeAffinity:
    required:
      nodeSelectorTerms:
        - matchExpressions:
           - key: kubernetes.io/hostname       or        - {key: kubernetes.io/hostname,operator: In, values: ["controlplane"]}
             operator: In
             values:
               - node1
